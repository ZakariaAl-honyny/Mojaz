```bash
# Backend Development Commands

# Create a new entity

src/backend/Mojaz.Domain/Entities/{EntityName}.cs
following the schema in PRD Section 21.
Also create the EF Core configuration in
src/backend/Mojaz.Infrastructure/Data/Configurations/{EntityName}Configuration.cs
Include all constraints, indexes, and relationships."

# Create a new service
opencode "Create the {Service}Service for Mojaz:
1. Interface I{Service}Service in Mojaz.Application/Interfaces/
2. Implementation {Service}Service in Mojaz.Application/Services/
3. Request/Response DTOs in Mojaz.Application/DTOs/
4. FluentValidation validators in Mojaz.Application/Validators/
5. AutoMapper profile in Mojaz.Application/Mappings/
Follow Clean Architecture rules from AGENTS.md.
All configurable values from SystemSettings table."

# Create a new controller
opencode "Create the {Resource}Controller in 
src/backend/Mojaz.API/Controllers/
with these endpoints: {list endpoints}
Follow rules:
- Return ApiResponse<T> always
- [Authorize(Roles = '{roles}')]  
- Swagger documentation
- CancellationToken on all async methods
- Call service methods only (no business logic)"

# Generate migration
opencode "Run: dotnet ef migrations add {MigrationName} 
--project src/backend/Mojaz.Infrastructure 
--startup-project src/backend/Mojaz.API
Then verify the migration looks correct."
```

### 2.3 Batch Scripts for OpenCode

```bash
#!/bin/bash
# scripts/opencode-batch-backend-entity.sh
# Usage: ./scripts/opencode-batch-backend-entity.sh EntityName

ENTITY=$1

echo "━━━ Creating $ENTITY entity, configuration, and repository ━━━"

opencode "Create the complete backend implementation for $ENTITY:

1. Entity: src/backend/Mojaz.Domain/Entities/$ENTITY.cs
   - Match PRD Section 21 schema exactly
   - Include navigation properties

2. Configuration: src/backend/Mojaz.Infrastructure/Data/Configurations/${ENTITY}Configuration.cs
   - Fluent API only
   - All constraints, indexes, relationships
   - String length limits

3. Add DbSet to AppDbContext

4. Repository: src/backend/Mojaz.Infrastructure/Repositories/${ENTITY}Repository.cs
   - Implement I${ENTITY}Repository
   - Include standard CRUD + soft delete

5. Interface: src/backend/Mojaz.Domain/Interfaces/I${ENTITY}Repository.cs

Follow all rules in AGENTS.md."
```

```bash
#!/bin/bash
```
```bash
# scripts/opencode-batch-feature.sh
# Usage: ./scripts/opencode-batch-feature.sh 011 application-crud

FEATURE_NUM=$1
FEATURE_NAME=$2

echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "  Starting Feature $FEATURE_NUM: $FEATURE_NAME"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

# Step 1: Read spec and create plan
opencode "Read the feature spec at features/$FEATURE_NUM-$FEATURE_NAME/
and create a detailed implementation plan:
1. List all files to create (in order)
2. List all files to modify
3. Identify dependencies
4. Estimate complexity per file
Save plan to features/$FEATURE_NUM-$FEATURE_NAME/plan.md"

echo "Plan created. Review and press Enter to continue..."
read

# Step 2: Implement backend
opencode "Implement all backend code for feature $FEATURE_NUM-$FEATURE_NAME
following the plan at features/$FEATURE_NUM-$FEATURE_NAME/plan.md
Create all: entities, services, DTOs, validators, controllers.
Follow AGENTS.md rules strictly."

echo "Backend done. Press Enter for frontend..."
read

# Step 3: Implement frontend
opencode "Implement all frontend code for feature $FEATURE_NUM-$FEATURE_NAME
following the plan at features/$FEATURE_NUM-$FEATURE_NAME/plan.md
Create all: pages, components, hooks, translations.
Follow AGENTS.md rules strictly."

echo "Frontend done. Press Enter for tests..."
read

# Step 4: Write tests
opencode "Write tests for feature $FEATURE_NUM-$FEATURE_NAME:
- Backend unit tests (xUnit)
- Frontend component tests (Jest) if applicable
Follow testing rules in AGENTS.md."

echo "━━━ Feature $FEATURE_NUM complete! ━━━"
```
